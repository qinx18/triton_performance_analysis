      n = b.shape[0]
      stride = 4
      start_idx = stride  # Loop starts at i=4
      
      # Process each stream independently
      for stream in range(stride):
          # Extract indices for this stream: stream, stream+stride, stream+2*stride, ...
          stream_indices = torch.arange(stream + stride, n, stride, device=b.device)
          if len(stream_indices) == 0:
              continue
          
          # Extract the addends for this stream
          addend_vals = a[stream_indices]
          
          # Compute prefix sum of addends
          prefix_sums = torch.cumsum(addend_vals, dim=0)
          
          # Add the initial value b[stream]
          b[stream_indices] = b[stream] + prefix_sums
